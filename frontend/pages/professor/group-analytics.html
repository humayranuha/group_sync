// Group Analytics Page import { renderProfessorLayout, initProfessorLayout } from '../../layouts/ProfessorLayout.js'; export function renderGroupAnalytics() { const content = `

Team Members Performance

Select a group to view member performance

🤖 AI Recommendations

`; const layout = renderProfessorLayout(content, 'analytics'); document.getElementById('app').innerHTML = layout; initProfessorLayout(); loadCourses(); setupEventListeners(); } let currentGroupData = null; async function loadCourses() { try { const user = JSON.parse(localStorage.getItem('groupsync_user_data') || '{}'); if (window.courseService) { const courses = await window.courseService.getCoursesByProfessor(user.id); const select = document.getElementById('course-select'); select.innerHTML = '' + courses.map(c => ``).join(''); } } catch (error) { console.error('Error loading courses:', error); window.toast?.error('Failed to load courses'); } } async function loadGroups(courseId) { try { if (window.groupService) { const groups = await window.groupService.getCourseGroups(courseId); const select = document.getElementById('group-select'); select.disabled = false; select.innerHTML = '' + groups.map(g => ``).join(''); } } catch (error) { console.error('Error loading groups:', error); window.toast?.error('Failed to load groups'); } } async function loadGroupAnalytics(groupId) { window.loading?.show(); try { if (window.analyticsService && window.groupService) { const analytics = await window.analyticsService.getGroupAnalytics(groupId); const members = await window.groupService.getGroupMembers(groupId); const group = await window.groupService.getGroupById(groupId); currentGroupData = { analytics, members, group }; updateGroupOverview(group, analytics, members); renderContributionChart(members); renderActivityTrendChart(analytics); renderMembersTable(members); renderAIRecommendations(members); document.getElementById('group-overview').classList.remove('hidden'); document.getElementById('ai-recommendations').classList.remove('hidden'); } } catch (error) { console.error('Error loading group analytics:', error); window.toast?.error('Failed to load group analytics'); } finally { window.loading?.hide(); } } function updateGroupOverview(group, analytics, members) { document.getElementById('group-name').textContent = group.name; document.getElementById('group-course').textContent = `Course: ${group.course_name || 'N/A'}`; document.getElementById('total-members').textContent = members?.length || 0; document.getElementById('total-commits').textContent = analytics.total_commits || 0; document.getElementById('avg-activity').textContent = `${analytics.avg_activity || 0}%`; const freeRiders = members?.filter(m => m.classification === 'Free Rider').length || 0; document.getElementById('free-riders-count').textContent = freeRiders; } function renderContributionChart(members) { const ctx = document.getElementById('contributionChart')?.getContext('2d'); if (!ctx || !members) return; new Chart(ctx, { type: 'doughnut', data: { labels: members.map(m => m.name), datasets: [{ data: members.map(m => m.contribution_percentage || 0), backgroundColor: ['#3b82f6', '#60a5fa', '#93c5fd', '#bfdbfe', '#1e3a8a', '#2563eb', '#1d4ed8', '#1e40af'], borderWidth: 0 }] }, options: { responsive: true, maintainAspectRatio: true, plugins: { legend: { position: 'bottom' }, tooltip: { callbacks: { label: (context) => { const label = context.label || ''; const value = context.parsed || 0; return `${label}: ${value}%`; } } } } } }); } function renderActivityTrendChart(analytics) { const ctx = document.getElementById('activityTrendChart')?.getContext('2d'); if (!ctx) return; new Chart(ctx, { type: 'line', data: { labels: analytics.weekly_labels || ['Week 1', 'Week 2', 'Week 3', 'Week 4', 'Week 5', 'Week 6', 'Week 7', 'Week 8'], datasets: [{ label: 'Team Activity Score', data: analytics.weekly_scores || [65, 68, 70, 72, 75, 78, 80, 82], borderColor: '#3b82f6', backgroundColor: 'rgba(59, 130, 246, 0.1)', tension: 0.4, fill: true }] }, options: { responsive: true, maintainAspectRatio: true, plugins: { legend: { position: 'top' } }, scales: { y: { beginAtZero: true, max: 100, title: { display: true, text: 'Activity Score (%)' } } } } }); } function renderMembersTable(members) { const container = document.getElementById('members-table'); if (members && members.length > 0) { container.innerHTML = ` ${members.map(member => ` `).join('')}
Student Commits PRs Lines Added Contribution % Classification Risk Level
${getInitials(member.name)}

${member.name}

${member.email || ''}

 ${member.commits || 0} ${member.prs || 0} ${(member.lines_added || 0).toLocaleString()}
${member.contribution_percentage || 0}%
${member.classification || 'N/A'} ${member.risk_level || 'Low'}
`; } else { container.innerHTML = '
No members found in this group
'; } } function renderAIRecommendations(members) { const container = document.getElementById('recommendations-list'); const freeRiders = members?.filter(m => m.classification === 'Free Rider') || []; const passiveMembers = members?.filter(m => m.classification === 'Passive') || []; let recommendations = []; if (freeRiders.length > 0) { recommendations.push(`⚠️ ${freeRiders.length} free rider(s) detected. Consider scheduling a meeting to discuss contribution expectations.`); } if (passiveMembers.length > 0) { recommendations.push(`📈 ${passiveMembers.length} passive member(s) identified. Consider pairing them with active members for mentoring.`); } if (members && members.length > 0) { const avgContribution = members.reduce((sum, m) => sum + (m.contribution_percentage || 0), 0) / members.length; if (avgContribution < 20) { recommendations.push(`🎯 Team contribution distribution is uneven. Consider restructuring tasks for better balance.`); } } if (recommendations.length === 0) { recommendations.push('✅ Team is performing well. Continue encouraging collaboration and regular contributions.'); } container.innerHTML = recommendations.map(rec => `

${rec}

`).join(''); } function getInitials(name) { if (!name) return '?'; return name.split(' ').map(n => n[0]).join('').toUpperCase().slice(0, 2); } function getClassificationClass(classification) { const classes = { 'Active': 'bg-green-100 text-green-800', 'Moderate': 'bg-yellow-100 text-yellow-800', 'Passive': 'bg-orange-100 text-orange-800', 'Free Rider': 'bg-red-100 text-red-800' }; return classes[classification] || 'bg-gray-100 text-gray-600'; } function getRiskClass(riskLevel) { const classes = { 'Low': 'bg-green-100 text-green-800', 'Medium': 'bg-yellow-100 text-yellow-800', 'High': 'bg-red-100 text-red-800' }; return classes[riskLevel] || 'bg-gray-100 text-gray-800'; } function setupEventListeners() { const courseSelect = document.getElementById('course-select'); const groupSelect = document.getElementById('group-select'); courseSelect?.addEventListener('change', (e) => { if (e.target.value) { loadGroups(e.target.value); } else { groupSelect.disabled = true; groupSelect.innerHTML = ''; document.getElementById('group-overview').classList.add('hidden'); document.getElementById('ai-recommendations').classList.add('hidden'); } }); groupSelect?.addEventListener('change', (e) => { if (e.target.value) { loadGroupAnalytics(e.target.value); } else { document.getElementById('group-overview').classList.add('hidden'); document.getElementById('ai-recommendations').classList.add('hidden'); } }); } export default renderGroupAnalytics;
