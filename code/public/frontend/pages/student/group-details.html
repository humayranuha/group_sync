// Group Details Page import { renderStudentLayout, initStudentLayout } from '../../layouts/StudentLayout.js'; export function renderGroupDetails() { const content = `

Loading...

Course: Loading...

0 Members

Total Contributions

0

Average Activity

0%

Team Performance

0/100

Team Members

Loading team members...

Contribution Distribution

`; const layout = renderStudentLayout(content, 'group'); document.getElementById('app').innerHTML = layout; initStudentLayout(); loadGroupDetails(); } async function loadGroupDetails() { try { const user = JSON.parse(localStorage.getItem('groupsync_user_data') || '{}'); // Get user's group if (window.groupService) { const groups = await window.groupService.getGroups(); const userGroup = groups.find(g => g.members?.includes(user.id)); if (userGroup) { await loadGroupData(userGroup.id); } else { showNoGroupMessage(); } } } catch (error) { console.error('Error loading group:', error); showNoGroupMessage(); } } async function loadGroupData(groupId) { try { // Load group details const group = await window.groupService.getGroupById(groupId); document.getElementById('group-name').textContent = group.name; document.getElementById('course-name').textContent = `Course: ${group.course_name || 'N/A'}`; document.getElementById('member-count').textContent = `${group.member_count || 0} Members`; // Load group analytics const analytics = await window.analyticsService.getGroupAnalytics(groupId); updateGroupStats(analytics); // Load members const members = await window.groupService.getGroupMembers(groupId); renderMembersList(members); // Render contribution chart renderContributionChart(members); } catch (error) { console.error('Error loading group data:', error); window.toast?.error('Failed to load group data'); } } function updateGroupStats(analytics) { document.getElementById('total-contributions').textContent = analytics.total_contributions || 0; document.getElementById('avg-activity').textContent = `${analytics.avg_activity || 0}%`; document.getElementById('team-performance').textContent = analytics.team_performance || 0; } function renderMembersList(members) { const container = document.getElementById('members-list'); if (members && members.length > 0) { container.innerHTML = members.map(member => `
${getInitials(member.name)}

${member.name}

${member.role || 'Student'}

${member.contribution_percentage || 0}%

Contributions

${member.classification || 'Moderate'}
`).join(''); } else { container.innerHTML = '
No members found in this group.
'; } } function renderContributionChart(members) { const ctx = document.getElementById('contributionChart')?.getContext('2d'); if (!ctx || !members || members.length === 0) return; new Chart(ctx, { type: 'pie', data: { labels: members.map(m => m.name), datasets: [{ data: members.map(m => m.contribution_percentage || 0), backgroundColor: ['#3b82f6', '#60a5fa', '#93c5fd', '#bfdbfe', '#1e3a8a'], borderWidth: 0 }] }, options: { responsive: true, maintainAspectRatio: true, plugins: { legend: { position: 'bottom' }, tooltip: { callbacks: { label: (context) => { const label = context.label || ''; const value = context.parsed || 0; return `${label}: ${value}%`; } } } } } }); } function getInitials(name) { if (!name) return '?'; return name.split(' ').map(n => n[0]).join('').toUpperCase().slice(0, 2); } function getClassificationClass(classification) { const classes = { 'Active': 'bg-green-100 text-green-800', 'Moderate': 'bg-yellow-100 text-yellow-800', 'Passive': 'bg-orange-100 text-orange-800', 'Free Rider': 'bg-red-100 text-red-800' }; return classes[classification] || 'bg-gray-100 text-gray-600'; } function showNoGroupMessage() { const container = document.getElementById('members-list'); document.getElementById('group-name').textContent = 'No Group Assigned'; document.getElementById('course-name').textContent = 'You are not part of any group yet'; document.getElementById('member-count').textContent = '0 Members'; if (container) { container.innerHTML = `
👥

No Group Assigned

You haven't been assigned to any group yet. Wait for your professor to add you to a group.

`; } } export default renderGroupDetails;
