// Student Dashboard Page import { renderStudentLayout, initStudentLayout } from '../../layouts/StudentLayout.js'; export function renderStudentDashboard() { const content = `

Welcome back, ${getStudentName()}! 👋

Here's your academic performance overview

Total Commits

0

Pull Requests

0

Activity Score

0

Team Rank

#0

Commit Trends

Weekly Activity

Contribution Heatmap

Loading heatmap data...

AI Performance Classification

Loading...

Loading AI evaluation...

Recent Activity

Loading recent activity...
`; const layout = renderStudentLayout(content, 'dashboard'); document.getElementById('app').innerHTML = layout; initStudentLayout(); // Load dashboard data loadDashboardData(); } function getStudentName() { const user = JSON.parse(localStorage.getItem('groupsync_user_data') || '{}'); return user.first_name || 'Student'; } async function loadDashboardData() { try { const user = JSON.parse(localStorage.getItem('groupsync_user_data') || '{}'); // Load analytics data if (window.analyticsService) { const analytics = await window.analyticsService.getStudentAnalytics(user.id, null, 4); updateStats(analytics); renderCharts(analytics); renderHeatmap(analytics); } // Load AI evaluation if (window.aiService) { const aiEval = await window.aiService.getStudentEvaluation(user.id, null); updateAICard(aiEval); } // Load recent activity if (window.githubService) { const activities = await window.githubService.getRepositories(); renderRecentActivity(activities); } } catch (error) { console.error('Error loading dashboard:', error); window.toast?.error('Failed to load dashboard data'); } } function updateStats(analytics) { document.getElementById('total-commits').textContent = analytics.total_commits || 0; document.getElementById('total-prs').textContent = analytics.total_prs || 0; document.getElementById('activity-score').textContent = analytics.activity_consistency_score || 0; document.getElementById('team-rank').textContent = `#${analytics.team_rank || 1}`; } function renderCharts(analytics) { // Commit Trends Chart const commitCtx = document.getElementById('commitTrendChart')?.getContext('2d'); if (commitCtx && analytics.weekly_data) { new Chart(commitCtx, { type: 'line', data: { labels: analytics.weekly_data.map(w => `Week ${w.week}`), datasets: [{ label: 'Commits', data: analytics.weekly_data.map(w => w.commits), borderColor: '#3b82f6', backgroundColor: 'rgba(59, 130, 246, 0.1)', tension: 0.4, fill: true }] }, options: { responsive: true, maintainAspectRatio: true, plugins: { legend: { position: 'top' } } } }); } // Weekly Activity Chart const activityCtx = document.getElementById('weeklyActivityChart')?.getContext('2d'); if (activityCtx && analytics.daily_activity) { new Chart(activityCtx, { type: 'bar', data: { labels: ['Mon', 'Tue', 'Wed', 'Thu', 'Fri', 'Sat', 'Sun'], datasets: [{ label: 'Contributions', data: analytics.daily_activity, backgroundColor: '#60a5fa', borderRadius: 8 }] }, options: { responsive: true, maintainAspectRatio: true, plugins: { legend: { position: 'top' } } } }); } } function renderHeatmap(analytics) { const container = document.getElementById('heatmap-container'); if (!container) return; if (analytics.heatmap_data && analytics.heatmap_data.length > 0) { container.innerHTML = `
${analytics.heatmap_data.map(day => `
${day.day}
${day.count}
`).join('')}
`; } else { container.innerHTML = '
No heatmap data available
'; } } function getHeatmapColor(count) { if (count === 0) return 'bg-gray-200 text-gray-600'; if (count < 5) return 'bg-green-200 text-green-800'; if (count < 10) return 'bg-green-400 text-white'; if (count < 20) return 'bg-green-600 text-white'; return 'bg-green-800 text-white'; } function updateAICard(aiEval) { const badge = document.getElementById('classification-badge'); const bar = document.getElementById('performance-bar'); const feedback = document.getElementById('ai-feedback-text'); if (aiEval) { const classification = aiEval.classification || 'Moderate'; const score = aiEval.overall_score || 75; badge.textContent = classification; badge.className = `px-3 py-1 rounded-full text-sm font-medium ${getClassificationClass(classification)}`; bar.style.width = `${score}%`; feedback.textContent = aiEval.feedback || 'Your performance is on track. Keep contributing!'; } else { badge.textContent = 'Not Available'; badge.className = 'px-3 py-1 rounded-full text-sm font-medium bg-gray-100 text-gray-600'; bar.style.width = '0%'; feedback.textContent = 'Connect your GitHub repository to receive AI feedback.'; } } function getClassificationClass(classification) { const classes = { 'Active': 'bg-green-100 text-green-800', 'Moderate': 'bg-yellow-100 text-yellow-800', 'Passive': 'bg-orange-100 text-orange-800', 'Free Rider': 'bg-red-100 text-red-800' }; return classes[classification] || 'bg-gray-100 text-gray-600'; } function renderRecentActivity(activities) { const container = document.getElementById('recent-activity-list'); if (!container) return; if (activities && activities.length > 0) { container.innerHTML = activities.slice(0, 5).map(activity => `
${getActivityIcon(activity.type)}

${activity.description}

${activity.repository_name} • ${getRelativeTime(activity.created_at)}

${activity.quantity || ''}
`).join(''); } else { container.innerHTML = '
No recent activity found. Connect your GitHub repository to start tracking.
'; } } function getActivityIcon(type) { const icons = { 'commit': '📝', 'pull_request': '🔄', 'fork': '🍴', 'branch': '🌿' }; return icons[type] || '📊'; } function getActivityIconBg(type) { const colors = { 'commit': 'bg-blue-100', 'pull_request': 'bg-green-100', 'fork': 'bg-purple-100', 'branch': 'bg-yellow-100' }; return colors[type] || 'bg-gray-100'; } function getRelativeTime(date) { if (!date) return 'Recently'; const now = new Date(); const diff = now - new Date(date); const minutes = Math.floor(diff / 60000); if (minutes < 1) return 'Just now'; if (minutes < 60) return `${minutes} min ago`; if (minutes < 1440) return `${Math.floor(minutes / 60)} hours ago`; return `${Math.floor(minutes / 1440)} days ago`; } export default renderStudentDashboard;
