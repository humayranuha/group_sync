// Repository Connection Page import { renderStudentLayout, initStudentLayout } from '../../layouts/StudentLayout.js'; export function renderRepositoryConnection() { const content = `

GitHub Repository Connection

🐙

Connect GitHub Repository

Connect your repository to start tracking contributions

Connect Repository Manually

Connected Repositories

Loading repositories...
`; const layout = renderStudentLayout(content, 'repository'); document.getElementById('app').innerHTML = layout; initStudentLayout(); loadRepositories(); setupEventListeners(); } async function loadRepositories() { const container = document.getElementById('repositories-list'); try { if (window.githubService) { const repos = await window.githubService.getRepositories(); if (repos && repos.length > 0) { container.innerHTML = repos.map(repo => `
📦

${repo.name}

${repo.url}

${repo.type} Last synced: ${formatDate(repo.last_synced_at)}
`).join(''); // Add event listeners for sync/remove buttons document.querySelectorAll('.sync-repo-btn').forEach(btn => { btn.addEventListener('click', () => syncRepository(btn.dataset.repoId)); }); document.querySelectorAll('.remove-repo-btn').forEach(btn => { btn.addEventListener('click', () => removeRepository(btn.dataset.repoId)); }); } else { container.innerHTML = `
🔗

No Repositories Connected

Connect your first GitHub repository to start tracking contributions.

`; } } } catch (error) { console.error('Error loading repositories:', error); container.innerHTML = `
Failed to load repositories. Please try again.
`; } } async function syncRepository(repoId) { window.loading?.show(); try { await window.githubService.syncRepository(repoId); window.toast?.success('Repository synced successfully'); await loadRepositories(); } catch (error) { window.toast?.error('Failed to sync repository'); } finally { window.loading?.hide(); } } async function removeRepository(repoId) { const confirmed = confirm('Are you sure you want to remove this repository?'); if (!confirmed) return; window.loading?.show(); try { await window.githubService.disconnectRepository(repoId); window.toast?.success('Repository removed successfully'); await loadRepositories(); } catch (error) { window.toast?.error('Failed to remove repository'); } finally { window.loading?.hide(); } } async function connectManualRepository() { const url = document.getElementById('manual-repo-url').value; const type = document.getElementById('manual-repo-type').value; if (!url) { window.toast?.warning('Please enter a repository URL'); return; } window.loading?.show(); try { await window.githubService.connectRepository(url, null, type); window.toast?.success('Repository connected successfully'); document.getElementById('manual-repo-url').value = ''; await loadRepositories(); } catch (error) { window.toast?.error('Failed to connect repository'); } finally { window.loading?.hide(); } } function setupEventListeners() { document.getElementById('connect-github-btn')?.addEventListener('click', () => { window.toast?.info('GitHub OAuth flow would initiate here'); }); document.getElementById('manual-connect-btn')?.addEventListener('click', connectManualRepository); document.getElementById('refresh-btn')?.addEventListener('click', loadRepositories); } function getRepoTypeBadge(type) { const badges = { 'original': 'bg-green-100 text-green-800', 'collaborator': 'bg-blue-100 text-blue-800', 'forked': 'bg-purple-100 text-purple-800' }; return badges[type] || 'bg-gray-100 text-gray-800'; } function formatDate(date) { if (!date) return 'Never'; const d = new Date(date); return d.toLocaleDateString(); } export default renderRepositoryConnection;
