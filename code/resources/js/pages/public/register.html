// Register Page export function renderRegisterPage() { const app = document.getElementById('app'); app.innerHTML = `
🚀

Join GroupSync

Start your academic collaboration journey

Create an account to access powerful analytics and AI insights

Create Account

Join GroupSync today

Optional for students

Must be at least 8 characters with uppercase, lowercase, and number

Or

Already have an account? Sign in

`; // Handle registration form submission const form = document.getElementById('register-form'); form.addEventListener('submit', async (e) => { e.preventDefault(); const password = document.getElementById('password').value; const confirmPassword = document.getElementById('confirm_password').value; if (password !== confirmPassword) { window.toast.error('Passwords do not match'); return; } const role = document.getElementById('role').value; const userData = { first_name: document.getElementById('first_name').value, last_name: document.getElementById('last_name').value, email: document.getElementById('email').value, role: role, password: password }; if (role === 'student') { userData.student_id = document.getElementById('student_id').value; } else if (role === 'professor') { userData.department = document.getElementById('department').value; } if (window.authService) { window.loading.show(); try { await window.authService.register(userData); window.toast.success('Registration successful! Please login.'); window.location.href = '/login.html'; } catch (error) { window.toast.error(error.message || 'Registration failed'); } finally { window.loading.hide(); } } }); // Handle GitHub registration const githubBtn = document.getElementById('github-register'); if (githubBtn) { githubBtn.addEventListener('click', () => { window.toast.info('GitHub OAuth redirect would happen here'); }); } // Show/hide fields based on role const roleSelect = document.getElementById('role'); const studentIdField = document.getElementById('student_id').closest('div'); const departmentField = document.getElementById('department').closest('div'); function toggleFields() { if (roleSelect.value === 'student') { studentIdField.style.display = 'block'; departmentField.style.display = 'none'; } else { studentIdField.style.display = 'none'; departmentField.style.display = 'block'; } } roleSelect.addEventListener('change', toggleFields); toggleFields(); } export default renderRegisterPage;
