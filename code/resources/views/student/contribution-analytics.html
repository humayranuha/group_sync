// Contribution Analytics Page import { renderStudentLayout, initStudentLayout } from '../../layouts/StudentLayout.js'; export function renderContributionAnalytics() { const content = `

Contribution Analytics

Total Commits

0

↑ 0%

Lines Added

0

Lines Deleted

0

Consistency Score

0

Commit Activity

Code Changes

Activity Consistency

Contribution Heatmap

Loading heatmap...

Top Repositories

Loading repositories...
`; const layout = renderStudentLayout(content, 'analytics'); document.getElementById('app').innerHTML = layout; initStudentLayout(); loadAnalytics(); setupTimeframeListener(); } let currentChart = {}; async function loadAnalytics(timeframe = 'week') { try { const user = JSON.parse(localStorage.getItem('groupsync_user_data') || '{}'); if (window.analyticsService) { const analytics = await window.analyticsService.getStudentAnalytics(user.id, null, timeframe); updateMetrics(analytics); renderCommitChart(analytics); renderCodeChangesChart(analytics); renderConsistencyChart(analytics); renderHeatmap(analytics); renderTopRepositories(analytics); } } catch (error) { console.error('Error loading analytics:', error); window.toast?.error('Failed to load analytics data'); } } function updateMetrics(analytics) { document.getElementById('total-commits').textContent = analytics.total_commits || 0; document.getElementById('lines-added').textContent = (analytics.total_lines_added || 0).toLocaleString(); document.getElementById('lines-deleted').textContent = (analytics.total_lines_deleted || 0).toLocaleString(); document.getElementById('consistency-score').textContent = analytics.consistency_score || 0; const trend = analytics.commit_trend || 0; const trendElement = document.getElementById('commit-trend'); if (trend >= 0) { trendElement.innerHTML = `↑ ${trend}%`; trendElement.className = 'text-xs text-green-600 mt-1'; } else { trendElement.innerHTML = `↓ ${Math.abs(trend)}%`; trendElement.className = 'text-xs text-red-600 mt-1'; } } function renderCommitChart(analytics) { const ctx = document.getElementById('commitChart')?.getContext('2d'); if (!ctx) return; if (currentChart.commit) currentChart.commit.destroy(); currentChart.commit = new Chart(ctx, { type: 'line', data: { labels: analytics.commit_labels || ['Week 1', 'Week 2', 'Week 3', 'Week 4'], datasets: [{ label: 'Commits', data: analytics.commit_data || [0, 0, 0, 0], borderColor: '#3b82f6', backgroundColor: 'rgba(59, 130, 246, 0.1)', tension: 0.4, fill: true }] }, options: { responsive: true, maintainAspectRatio: true, plugins: { legend: { position: 'top' } } } }); } function renderCodeChangesChart(analytics) { const ctx = document.getElementById('codeChangesChart')?.getContext('2d'); if (!ctx) return; if (currentChart.codeChanges) currentChart.codeChanges.destroy(); currentChart.codeChanges = new Chart(ctx, { type: 'bar', data: { labels: analytics.commit_labels || ['Week 1', 'Week 2', 'Week 3', 'Week 4'], datasets: [ { label: 'Lines Added', data: analytics.lines_added_data || [0, 0, 0, 0], backgroundColor: '#10b981' }, { label: 'Lines Deleted', data: analytics.lines_deleted_data || [0, 0, 0, 0], backgroundColor: '#ef4444' } ] }, options: { responsive: true, maintainAspectRatio: true, plugins: { legend: { position: 'top' } } } }); } function renderConsistencyChart(analytics) { const ctx = document.getElementById('consistencyChart')?.getContext('2d'); if (!ctx) return; if (currentChart.consistency) currentChart.consistency.destroy(); currentChart.consistency = new Chart(ctx, { type: 'radar', data: { labels: ['Commits', 'PRs', 'Code Reviews', 'Documentation', 'Testing', 'Issues'], datasets: [{ label: 'Your Activity', data: analytics.radar_data || [85, 70, 60, 75, 65, 80], borderColor: '#3b82f6', backgroundColor: 'rgba(59, 130, 246, 0.2)' }] }, options: { responsive: true, maintainAspectRatio: true, scales: { r: { beginAtZero: true, max: 100 } } } }); } function renderHeatmap(analytics) { const container = document.getElementById('heatmap-container'); if (!container) return; if (analytics.heatmap_data && analytics.heatmap_data.length > 0) { container.innerHTML = `
${analytics.heatmap_data.map(day => `
${day.day}
${day.count}
`).join('')}
`; } else { container.innerHTML = '
No heatmap data available
'; } } function getHeatmapColor(count) { if (count === 0) return 'bg-gray-200 text-gray-600'; if (count < 5) return 'bg-green-200 text-green-800'; if (count < 10) return 'bg-green-400 text-white'; if (count < 20) return 'bg-green-600 text-white'; return 'bg-green-800 text-white'; } function renderTopRepositories(analytics) { const container = document.getElementById('top-repos-list'); if (analytics.top_repos && analytics.top_repos.length > 0) { container.innerHTML = analytics.top_repos.map(repo => `

${repo.name}

${repo.commits} commits • ${repo.lines_added} lines added

${repo.contribution_percentage}%

of total

`).join(''); } else { container.innerHTML = '
No repository data available
'; } } function setupTimeframeListener() { const select = document.getElementById('timeframe-select'); select?.addEventListener('change', (e) => { loadAnalytics(e.target.value); }); } export default renderContributionAnalytics;
