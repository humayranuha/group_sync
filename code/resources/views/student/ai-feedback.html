`; const layout = renderStudentLayout(content, 'ai-feedback'); document.getElementById('app').innerHTML = layout; initStudentLayout(); loadAIFeedback(); setupRefreshButton(); } let feedbackCharts = {}; async function loadAIFeedback() { try { const user = JSON.parse(localStorage.getItem('groupsync_user_data') || '{}'); if (window.aiService) { const feedback = await window.aiService.getStudentEvaluation(user.id, null); updateAIFeedback(feedback); renderWeeklyTrendChart(feedback); renderTeamComparisonChart(feedback); } } catch (error) { console.error('Error loading AI feedback:', error); window.toast?.error('Failed to load AI feedback'); } } function updateAIFeedback(feedback) { // Update classification badge const badge = document.getElementById('classification-badge'); const classification = feedback.classification || 'Moderate'; badge.textContent = classification; badge.className = `px-4 py-2 rounded-full text-lg font-semibold ${getClassificationClass(classification)}`; // Update scores document.getElementById('participation-score').textContent = feedback.participation_score || 0; document.getElementById('quality-score').textContent = feedback.quality_score || 0; document.getElementById('consistency-score').textContent = feedback.consistency_score || 0; // Update feedback text const feedbackText = document.getElementById('feedback-text'); feedbackText.textContent = feedback.feedback || 'No detailed feedback available at this time.'; // Update suggestions const suggestionsList = document.getElementById('suggestions-list'); if (feedback.suggestions && feedback.suggestions.length > 0) { suggestionsList.innerHTML = feedback.suggestions.map(suggestion => `
  • • ${suggestion}
    • `).join(''); } else { suggestionsList.innerHTML = '
  • No specific suggestions available.
    • '; } // Show suspicious activity if detected if (feedback.suspicious_activity_detected) { const suspiciousCard = document.getElementById('suspicious-card'); suspiciousCard.classList.remove('hidden'); document.getElementById('suspicious-text').textContent = feedback.suspicious_reason || 'Unusual activity pattern detected. Please review your contribution history.'; } } function renderWeeklyTrendChart(feedback) { const ctx = document.getElementById('weeklyTrendChart')?.getContext('2d'); if (!ctx) return; if (feedbackCharts.weekly) feedbackCharts.weekly.destroy(); feedbackCharts.weekly = new Chart(ctx, { type: 'line', data: { labels: feedback.weekly_labels || ['Week 1', 'Week 2', 'Week 3', 'Week 4', 'Week 5', 'Week 6', 'Week 7', 'Week 8'], datasets: [ { label: 'Your Performance', data: feedback.weekly_scores || [70, 72, 75, 78, 80, 82, 85, 87], borderColor: '#3b82f6', backgroundColor: 'rgba(59, 130, 246, 0.1)', tension: 0.4, fill: true }, { label: 'Team Average', data: feedback.team_weekly_scores || [68, 70, 71, 73, 74, 75, 76, 77], borderColor: '#10b981', backgroundColor: 'rgba(16, 185, 129, 0.1)', tension: 0.4, fill: true } ] }, options: { responsive: true, maintainAspectRatio: true, plugins: { legend: { position: 'top' }, tooltip: { callbacks: { label: (context) => { return `${context.dataset.label}: ${context.parsed.y}%`; } } } }, scales: { y: { beginAtZero: true, max: 100, title: { display: true, text: 'Performance Score (%)' } } } } }); } function renderTeamComparisonChart(feedback) { const ctx = document.getElementById('teamComparisonChart')?.getContext('2d'); if (!ctx) return; if (feedbackCharts.comparison) feedbackCharts.comparison.destroy(); feedbackCharts.comparison = new Chart(ctx, { type: 'radar', data: { labels: feedback.comparison_labels || ['Commits', 'Code Quality', 'PRs', 'Reviews', 'Documentation', 'Collaboration'], datasets: [ { label: 'You', data: feedback.user_metrics || [85, 78, 82, 70, 75, 80], borderColor: '#3b82f6', backgroundColor: 'rgba(59, 130, 246, 0.2)' }, { label: 'Team Average', data: feedback.team_metrics || [75, 72, 70, 68, 65, 70], borderColor: '#10b981', backgroundColor: 'rgba(16, 185, 129, 0.2)' } ] }, options: { responsive: true, maintainAspectRatio: true, scales: { r: { beginAtZero: true, max: 100, ticks: { stepSize: 20 } } } } }); } function getClassificationClass(classification) { const classes = { 'Active': 'bg-green-100 text-green-800', 'Moderate': 'bg-yellow-100 text-yellow-800', 'Passive': 'bg-orange-100 text-orange-800', 'Free Rider': 'bg-red-100 text-red-800' }; return classes[classification] || 'bg-gray-100 text-gray-600'; } function setupRefreshButton() { const refreshBtn = document.getElementById('refresh-feedback'); refreshBtn?.addEventListener('click', async () => { window.loading?.show(); await loadAIFeedback(); window.loading?.hide(); window.toast?.success('AI feedback refreshed'); }); } export default renderAIFeedback;
