// Professor Dashboard Page import { renderProfessorLayout, initProfessorLayout } from '../../layouts/ProfessorLayout.js'; export function renderProfessorDashboard() { const content = `

Welcome back, Professor ${getProfessorName()}! 👋

Here's your academic overview

Active Courses

0

Total Students

0

Student Groups

0

Free Riders

0

AI Risk Indicators by Course

Active Courses

View All →
Loading courses...

Recent Alerts

No recent alerts
`; const layout = renderProfessorLayout(content, 'dashboard'); document.getElementById('app').innerHTML = layout; initProfessorLayout(); loadDashboardData(); } function getProfessorName() { const user = JSON.parse(localStorage.getItem('groupsync_user_data') || '{}'); return user.first_name || 'Professor'; } async function loadDashboardData() { try { const user = JSON.parse(localStorage.getItem('groupsync_user_data') || '{}'); if (window.courseService) { const courses = await window.courseService.getCoursesByProfessor(user.id); updateStats(courses); renderCoursesList(courses); renderRiskChart(courses); loadAlerts(courses); } } catch (error) { console.error('Error loading dashboard:', error); window.toast?.error('Failed to load dashboard data'); } } function updateStats(courses) { document.getElementById('active-courses').textContent = courses?.length || 0; let totalStudents = 0; let totalGroups = 0; courses?.forEach(course => { totalStudents += course.student_count || 0; totalGroups += course.group_count || 0; }); document.getElementById('total-students').textContent = totalStudents; document.getElementById('total-groups').textContent = totalGroups; document.getElementById('free-riders').textContent = Math.floor(totalStudents * 0.05); // Example calculation } function renderCoursesList(courses) { const container = document.getElementById('recent-courses-list'); if (courses && courses.length > 0) { container.innerHTML = courses.slice(0, 5).map(course => `
📚

${course.code} - ${course.name}

${course.semester} • ${course.student_count || 0} students • ${course.group_count || 0} groups

View →
`).join(''); } else { container.innerHTML = `
📖

No Courses Yet

Create your first course to get started.

Create Course
`; } } function renderRiskChart(courses) { const ctx = document.getElementById('riskChart')?.getContext('2d'); if (!ctx || !courses || courses.length === 0) return; new Chart(ctx, { type: 'bar', data: { labels: courses.map(c => c.code), datasets: [ { label: 'High Risk', data: courses.map(c => c.high_risk_count || 0), backgroundColor: '#ef4444' }, { label: 'Medium Risk', data: courses.map(c => c.medium_risk_count || 0), backgroundColor: '#f59e0b' }, { label: 'Low Risk', data: courses.map(c => c.low_risk_count || 0), backgroundColor: '#10b981' } ] }, options: { responsive: true, maintainAspectRatio: true, plugins: { legend: { position: 'top' }, tooltip: { callbacks: { label: (context) => `${context.dataset.label}: ${context.parsed.y} students` } } }, scales: { x: { stacked: true }, y: { stacked: true, title: { display: true, text: 'Number of Students' } } } } }); } async function loadAlerts(courses) { const container = document.getElementById('recent-alerts'); if (!courses || courses.length === 0) { container.innerHTML = '
No recent alerts
'; return; } try { let allAlerts = []; for (const course of courses) { if (window.aiService) { const alerts = await window.aiService.detectFreeRiders(course.id); if (alerts && alerts.length > 0) { allAlerts.push(...alerts.map(alert => ({ ...alert, course_name: course.name }))); } } } if (allAlerts.length > 0) { container.innerHTML = allAlerts.slice(0, 5).map(alert => `
${getAlertIcon(alert.severity)}

${alert.student_name}

${alert.course_name} • ${alert.message}

${alert.severity}
`).join(''); } else { container.innerHTML = '
No recent alerts
'; } } catch (error) { console.error('Error loading alerts:', error); container.innerHTML = '
Unable to load alerts
'; } } function getAlertColor(severity) { const colors = { 'High': 'bg-red-100', 'Medium': 'bg-yellow-100', 'Low': 'bg-green-100' }; return colors[severity] || 'bg-gray-100'; } function getAlertIcon(severity) { const icons = { 'High': '⚠️', 'Medium': '⚡', 'Low': 'ℹ️' }; return icons[severity] || '📌'; } function getSeverityClass(severity) { const classes = { 'High': 'bg-red-100 text-red-800', 'Medium': 'bg-yellow-100 text-yellow-800', 'Low': 'bg-green-100 text-green-800' }; return classes[severity] || 'bg-gray-100 text-gray-800'; } export default renderProfessorDashboard;
